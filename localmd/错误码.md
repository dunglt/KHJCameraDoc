# 错误码

```
error_Code = -20000，		// 该函数的传入参数不正确
error_Code = -20019，		// AV模块尚未初始化
error_Code = -20002，		// AV通道数已达到最大值
error_Code = -20004，		// AV无法创建线程
error_Code = -20007，		// 指定的AV服务器没有响应
error_Code = -20015，		// 远程站点已关闭
error_Code = -20016，		// 此会话已断开，因为远程站点在指定的超时到期后没有任何响应
error_Code = -20010，		// 指定AV通道的会话无效
error_Code = -20018，		// 客户端启动过程终止
error_Code = -20011，		// 某些操作期间指定的超时已到期
error_Code = -20009，		// 由于查看帐户或密码不正确，客户端验证失败
error_Code = -20023，		// UID是精简版UID
error_Code = -10，		   // 指定的UID未获得许可或已过期
error_Code = -90，    	   // 所有服务器响应都找不到设备
```
